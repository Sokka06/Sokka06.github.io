import {
  Tag
} from '@chakra-ui/react'

export function Age (dob) {
    const diff_ms = Date.now() - dob.getTime()
    const age_dt = new Date(diff_ms)
    return Math.abs(age_dt.getUTCFullYear() - 1970)
}
